class Solution {
public:
    bool isPalindrome(string s) {
        string res="";
        string res1;
        for(int i=0;i<s.size();i++)
        {
            if(isalnum(s[i]))
            {
                res+=tolower(s[i]);
            }
        }
        res1=res;
        reverse(res.begin(),res.end());
        return res1==res;
    }
};
